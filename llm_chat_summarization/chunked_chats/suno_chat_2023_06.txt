# %%
# Generate some initial samples
from audiocraft.utils.notebook import display_audio

model.set_generation_params(
    duration=30.0
)

descriptions = [
    "Twirling melodies, harmonious vocals, and infectious beats transport you to a groovy world of love and peace in this 60s pop song.",
    "An ethereal fusion of classical elegance and whimsical surrealism, this song inspired by Mozart enchants the senses with its intricate melodies and dreamlike imagery.",
    "An enchanting blend of intricate baroque arrangements and surrealistic elements, this Bach-inspired song takes you on a journey through a kaleidoscope of musical wonders that transcend time and space.",
    "Surfing through a kaleidoscope of mind-bending sounds, this psychedelic song rides the waves of a reverb-drenched guitar, swirling organs, and trippy lyrics, immersing you in a sonic journey that transcends reality."
]

output = model.generate(
    descriptions=descriptions,
    progress=True
)
display_audio(output, sample_rate=32000)

# %%
second = int(output.shape[-1] / 30)
n_seconds = second * 20
# %%
# Repeat this part as many times as you want.
output_full = output
for _ in range(10):
  output_prev = output_full[:, :, -n_seconds:]
  output_next = model.generate_continuation(output_prev, 32000, descriptions, progress=True)
  output_full = torch.cat([output_full[:, :, :-n_seconds], output_next], dim=-1)
display_audio(output_full, sample_rate=32000)
```
